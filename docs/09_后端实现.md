## 1. 启动流程

`main.py` 按顺序执行：

```python
# main.py
from contextlib import asynccontextmanager
from fastapi import FastAPI

@asynccontextmanager
async def lifespan(app: FastAPI):
    await config.load()                  # 1. 加载 .env → Config 对象
    await database.init_postgres()       # 2. 创建 PG 连接池（asyncpg）
    await database.init_sqlite()         # 3. 初始化 SQLite food_db（含 FTS5 索引）
    await database.init_elasticsearch()  # 4. ES 连接，检查索引是否存在
    await database.init_redis()          # 5. Redis 连接
    await index_service.ensure_index()   # 6. 确保 ES 索引存在（不存在则创建空索引）
    yield
    await database.close_all()           # 关闭所有连接池

app = FastAPI(lifespan=lifespan)

# 注册路由
app.include_router(auth_router, prefix="/api/auth")
app.include_router(diet_router, prefix="/api/diet")
app.include_router(food_router, prefix="/api/food")
app.include_router(search_router, prefix="/api/search")
app.include_router(restaurants_router, prefix="/api/restaurants")
app.include_router(feedback_router, prefix="/api/feedback")
app.include_router(internal_router, prefix="/api/internal")
```

启动命令：
```bash
uvicorn main:app --host 0.0.0.0 --port 8000 --reload   # 开发
uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4 # 生产
```

## 3. 数据库层

### 3.1 PostgreSQL（asyncpg + SQLAlchemy 2.0）

```python
# database.py
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession

engine = create_async_engine(
    config.postgres_dsn,
    pool_size=10,
    max_overflow=20,
    pool_pre_ping=True,   # 自动检测断开连接
)

async def get_db() -> AsyncSession:
    async with AsyncSession(engine) as session:
        yield session
```

**连接池参数：** `pool_size=10`，`max_overflow=20`，`pool_recycle=3600`

### 3.2 SQLite（aiosqlite）

```python
# database.py
import aiosqlite

_sqlite_conn: aiosqlite.Connection = None

async def init_sqlite():
    global _sqlite_conn
    _sqlite_conn = await aiosqlite.connect(config.sqlite_path)
    _sqlite_conn.row_factory = aiosqlite.Row
    # 启用 WAL 模式（提高并发读性能）
    await _sqlite_conn.execute("PRAGMA journal_mode=WAL")
    await _sqlite_conn.execute("PRAGMA foreign_keys=ON")
    await _sqlite_conn.commit()
```

**注意：** SQLite 为只读工作负载（食物搜索），写操作仅在食物确认入库时触发，并发写冲突风险极低。

## 4. 鉴权实现

```python
# middleware/auth_middleware.py
from fastapi import Request, HTTPException, Depends
from starlette.middleware.sessions import SessionMiddleware

# 在 app 注册时挂载
app.add_middleware(SessionMiddleware, secret_key=config.session_secret)

async def get_current_user(request: Request, db: AsyncSession = Depends(get_db)):
    """FastAPI Dependency，注入到需要登录的路由"""
    user_id = request.session.get("user_id")
    if not user_id:
        raise HTTPException(status_code=401, detail="请先登录")
    user = await db.get(User, user_id)
    if not user:
        request.session.clear()
        raise HTTPException(status_code=401, detail="用户不存在")
    return user

# 可选鉴权（匿名可访问，登录后有额外功能）
async def get_optional_user(request: Request, db: AsyncSession = Depends(get_db)):
    user_id = request.session.get("user_id")
    if not user_id:
        return None
    return await db.get(User, user_id)
```

## 5. IR 核心模块实现

### 5.1 查询解析器

```python
# ir/query_parser.py
import jieba
import spacy
from .synonyms import DietSynonymDict

class QueryParser:
    def __init__(self):
        self.nlp_en = spacy.load("en_core_web_sm")
        self.synonyms = DietSynonymDict.load("data/diet_synonyms.json")

    def parse(self, query: str) -> ParsedQuery:
        lang = detect_language(query)     # 语言检测

        if lang == "zh":
            tokens = list(jieba.cut_for_search(query))
        else:
            doc = self.nlp_en(query.lower())
            tokens = [token.lemma_ for token in doc
                      if not token.is_stop and not token.is_punct]

        # 饮食标签意图识别
        detected_diet_labels = []
        expanded_tokens = []
        for token in tokens:
            if token in self.synonyms:
                canonical, syns = self.synonyms[token]
                detected_diet_labels.append(canonical)
                expanded_tokens.extend(syns)  # 同义词扩展
            else:
                expanded_tokens.append(token)

        return ParsedQuery(
            original=query,
            tokens=tokens,
            expanded_tokens=expanded_tokens,
            detected_diet_labels=list(set(detected_diet_labels)),
            language=lang,
        )
```

### 5.2 排序服务

```python
# services/ranking_service.py
import yaml
from dataclasses import dataclass

class RankingService:
    def __init__(self, config_path: str):
        with open(config_path) as f:
            self._weights = yaml.safe_load(f)["sort_modes"]

    def get_weights(self, sort_mode: str) -> dict:
        return self._weights.get(sort_mode, self._weights["default"])

    def rerank(self, hits: list[dict], query: ParsedQuery,
               user_profile: DietProfile | None, user_geo: tuple | None) -> list[dict]:
        weights = self.get_weights(query.sort_mode)

        for hit in hits:
            r = hit["_source"]
            text_score = hit["_score"]  # ES BM25

            diet_score = calc_diet_match_score(
                r, query.detected_diet_labels,
                user_profile.allergens if user_profile else []
            )
            rating_score = r.get("rating", 0) / 5.0
            distance_score = calc_distance_score(r.get("geo"), user_geo)
            personalization = calc_personalization(r, user_profile)

            hit["_final_score"] = (
                weights["text_score"] * text_score +
                weights["diet_score"] * diet_score +
                weights["rating_score"] * rating_score +
                weights["distance_score"] * distance_score +
                weights["personalization"] * personalization
            )
            # 过敏原标记（BR-08）
            hit["allergen_warning"] = build_allergen_warning(
                r, user_profile.allergens if user_profile else []
            )

        return sorted(hits, key=lambda h: h["_final_score"], reverse=True)
```

### 5.3 食物服务（LLM fallback）

```python
# services/food_service.py
import json
import hashlib
from openai import AsyncOpenAI   # DeepSeek 兼容 OpenAI SDK

class FoodService:
    def __init__(self, db: aiosqlite.Connection, redis: Redis, llm_client: AsyncOpenAI):
        self.db = db
        self.redis = redis
        self.llm = llm_client

    async def search(self, query: str, limit: int = 10) -> SearchResult:
        # Step 1: SQLite FTS
        async with self.db.execute(
            "SELECT * FROM food_items JOIN food_fts ON food_items.id = food_fts.rowid "
            "WHERE food_fts MATCH ? AND verified = 1 "
            "ORDER BY rank LIMIT ?",
            (f"{query}*", limit)
        ) as cursor:
            rows = await cursor.fetchall()

        if rows:
            return SearchResult(source="database", items=rows, requires_confirm=False)

        # Step 2: Redis 缓存检查
        cache_key = f"food:llm:{hashlib.md5(query.encode()).hexdigest()}"
        cached = await self.redis.get(cache_key)
        if cached:
            return SearchResult(source="llm_estimated",
                                items=[json.loads(cached)],
                                requires_confirm=True)

        # Step 3: LLM 查询
        result = await self._query_llm(query)
        if result:
            await self.redis.set(cache_key, json.dumps(result), ex=604800)  # 7天
            return SearchResult(source="llm_estimated",
                                items=[result],
                                requires_confirm=True)

        return SearchResult(source="not_found", items=[], requires_confirm=False)

    async def _query_llm(self, food_name: str) -> dict | None:
        try:
            resp = await self.llm.chat.completions.create(
                model=config.llm_model,
                messages=[
                    {"role": "system", "content": "你是营养数据库助手，只返回 JSON。"},
                    {"role": "user", "content": LLM_PROMPT.format(food_name=food_name)}
                ],
                response_format={"type": "json_object"},
                timeout=10,
            )
            data = json.loads(resp.choices[0].message.content)
            if data.get("calories", -1) < 0:
                return None
            return data
        except Exception as e:
            logger.warning(f"LLM query failed for '{food_name}': {e}")
            return None
```

### 6.1 高德 POI 爬虫

```python
# crawler/gaode_crawler.py
import httpx

GAODE_PLACE_URL = "https://restapi.amap.com/v3/place/text"

class GaodeCrawler:
    async def fetch_all(self, city: str, limit: int = 500) -> list[RestaurantDoc]:
        results = []
        page = 1
        async with httpx.AsyncClient() as client:
            while len(results) < limit:
                resp = await client.get(GAODE_PLACE_URL, params={
                    "key": config.gaode_api_key,
                    "keywords": "餐厅|饭店|餐馆",
                    "city": city,
                    "page": page,
                    "offset": 20,
                    "extensions": "all",
                })
                data = resp.json()
                if data["status"] != "1" or not data["pois"]:
                    break
                for poi in data["pois"]:
                    results.append(self._normalize(poi))
                page += 1
                await asyncio.sleep(0.1)  # 高德官方 API 不限速，但保留间隔
        return results[:limit]
```

### 6.2 大众点评爬虫（Playwright）

```python
# crawler/dianping_crawler.py
from playwright.async_api import async_playwright

class DianpingCrawler:
    RATE_LIMIT = 3.0   # 秒

    async def enrich(self, restaurants: list[RestaurantDoc]):
        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            context = await browser.new_context(
                user_agent=random.choice(UA_POOL)
            )
            for r in restaurants:
                await self._fetch_one(context, r)
                await asyncio.sleep(self.RATE_LIMIT +
                                    random.uniform(0, 1))   # 随机抖动
            await browser.close()

    async def _fetch_one(self, context, restaurant: RestaurantDoc):
        page = await context.new_page()
        try:
            await page.goto(
                f"https://www.dianping.com/search/keyword/{restaurant.city}/10_{restaurant.name}",
                timeout=15000
            )
            # 提取第一个搜索结果的菜品和标签
            menu_items = await page.eval_on_selector_all(
                ".menu-item", "els => els.map(e => e.textContent)"
            )
            restaurant.menu_items = parse_menu_items(menu_items)
        except Exception as e:
            logger.warning(f"Dianping fetch failed for {restaurant.name}: {e}")
        finally:
            await page.close()
```

## 7. 错误处理规范

```python
# 全局异常处理器
@app.exception_handler(HTTPException)
async def http_exception_handler(request, exc):
    return JSONResponse(
        status_code=exc.status_code,
        content={"code": exc.status_code, "error": exc.detail}
    )

@app.exception_handler(Exception)
async def generic_exception_handler(request, exc):
    logger.error(f"Unhandled exception: {exc}", exc_info=True)
    return JSONResponse(
        status_code=500,
        content={"code": 500, "error": "服务器内部错误"}
    )
```

**原则：**
- 业务层抛出 `HTTPException`（含 status_code 和 detail）
- 服务层内部错误记录日志后抛出，不向用户暴露内部细节
- LLM/爬虫异常：降级处理，不影响主流程
